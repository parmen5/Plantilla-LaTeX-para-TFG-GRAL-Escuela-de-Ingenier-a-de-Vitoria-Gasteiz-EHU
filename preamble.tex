\newcommand{\portada}{
    \begin{titlepage}
        \begin{tikzpicture}[remember picture,overlay]
            \node[anchor=north] at ([yshift=-3cm]current page.north) {
                % Dejamos solo el nombre porque ya busca en la carpeta images
                \includegraphics[width=10cm]{ehu.png}
            };
            \node at ([yshift=5.5cm]current page.center) [text width=15cm, align=center] {
                {\Large \textbf{\carrera}} \\[1.2cm]
            };
            \node at ([yshift=1cm]current page.center) [text width=16cm, align=center] {
                {\huge \textbf{\titulo}}
            };
            \node[anchor=south] at ([yshift=3cm]current page.south) {
                \begin{tabular}{|p{7.5cm} p{7.5cm}|}
                    \hline
                    \multicolumn{2}{|l|}{} \\
                    \multicolumn{2}{|l|}{\large \textbf{Estudiante:} \alumno} \\
                    \multicolumn{2}{|l|}{} \\
                    \multicolumn{2}{|l|}{\large \textbf{Director/a:} \direccionA}  \\
                    \multicolumn{2}{|l|}{} \\ \hline
                    & \\
                    \large \textbf{Curso:} \curso & \large \textbf{Fecha:} \fecha  \\
                    & \\ \hline
                \end{tabular}
            };
        \end{tikzpicture}
    \end{titlepage}
}
